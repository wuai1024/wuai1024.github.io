                    # 1.1 查看防火墙状态

                    sudo ufw status
                    1

                    # 1.2 启用防火墙

                    注意：在启用的时候，一定要注意开放了ssh连接端口（例如：22），否则开启后可能出现连接不上的情况。

                    sudo ufw enable
                    1

                    # 1.3 配置默认规则

                    默认希望拒绝所有入站请求，允许所有出站请求。

                    sudo ufw default deny incoming
sudo ufw default allow outgoing
                    1
                    2

                    # 1.4 添加入站规则

                    sudo ufw allow 22/tcp  # 允许 SSH 端口
sudo ufw allow 443/tcp  # 允许 HTTPS 端口
                    1
                    2

                    # 1.5 删除入站规则

                    sudo ufw delete allow 22/tcp  # 删除允许 SSH 端口的规则
sudo ufw delete allow 443/tcp  # 删除允许 HTTPS 端口的规则
                    1
                    2

                    # 1.6 查看防火墙状态和规则

                    sudo ufw status verbose
