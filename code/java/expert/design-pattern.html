设计模式（开发高频）

1、策略模式

interface PayStrategy { void pay(int amount); }
class AliPayStrategy implements PayStrategy { public void pay(int amount){ System.out.println("alipay " + amount); } }
class WxPayStrategy implements PayStrategy { public void pay(int amount){ System.out.println("wxpay " + amount); } }

class PayContext {
    private final java.util.Map<String, PayStrategy> map;
    PayContext() {
        map = java.util.Map.of("ALI", new AliPayStrategy(), "WX", new WxPayStrategy());
    }
    void execute(String type, int amount) { map.get(type).pay(amount); }
}

2、模板方法

abstract class ExportTemplate {
    public final void export() {
        load();
        convert();
        save();
    }
    protected abstract void load();
    protected abstract void convert();
    protected abstract void save();
}
